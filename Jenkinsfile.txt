pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                 bat 'echo "Hello World outside retry and timeout"'
                retry(3) {
                    bat 'echo "Hello World inside retry"'
                }

                timeout(time: 1, unit: 'MINUTES') {
                    bat 'echo "Hello World2 inside timeout"'
                }
            }
        }
    }
}
